#include <unistd.h>

void    ft_putnbr(int nb)
{
    int a;
    int b;
    
    if (nb <  0)
    {
        write(1, "-", 1);
        nb = -nb;
        ft_putnbr(nb / 10);
        ft_putnbr(nb % 10);
    }
    else if(nb > 9)
    {
        ft_putnbr(nb / 10);
        ft_putnbr(nb % 10);
    }
    else
    {
        a = nb + '0';
        write(1, &a, 1);
    }
}

int main(void)
{
    ft_putnbr(-6767);
}
